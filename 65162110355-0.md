export function updateStreakState(currentState: StreakState, wasRoundWin: boolean): StreakState {
  if (!currentState.bonusTimeActive) {
    if (wasRoundWin) {
      const nextConsecutiveWins = currentState.consecutiveWins + 1;
      if (nextConsecutiveWins >= 3) {
        return {
          consecutiveWins: nextConsecutiveWins,
          bonusTimeActive: true,
          bonusTimeMultiplier: 2,
        };
      }
      return {
        ...currentState,
        consecutiveWins: nextConsecutiveWins,
      };
    } else {
      return {
        consecutiveWins: 0,
        bonusTimeActive: false,
        bonusTimeMultiplier: 0,
      };
    }
  } else {
    // Already in Bonus Time
    if (wasRoundWin) {
      return {
        consecutiveWins: currentState.consecutiveWins + 1,
        bonusTimeActive: true,
        bonusTimeMultiplier: Math.min(currentState.bonusTimeMultiplier + 1, 6),
      };
    } else {
      // Lose during bonus time -> reset completely
      return {
        consecutiveWins: 0,
        bonusTimeActive: false,
        bonusTimeMultiplier: 0,
      };
    }
  }
}

export function resolveMultiplierSource(
  streakState: StreakState,
  roundBonus: RoundBonus,
  bet: Bet,
  diceResult: DiceResult,
  symbol: GameSymbol
): { source: 'bonusTime' | 'roundBonus' | 'none'; multiplier: number } {
  // Rule: Round Bonus and Bonus Time are mutually exclusive.
  // If Bonus Time is active, Round Bonus is completely ignored.
  if (streakState.bonusTimeActive) {
    return {
      source: 'bonusTime',
      multiplier: streakState.bonusTimeMultiplier,
    };
  }

  // Check if this symbol was bet on and has a match
  const betAmount = bet.get(symbol) || 0;
  const matchCount = getMatchCount(diceResult, symbol);
  
  if (betAmount > 0 && matchCount > 0) {
    const bonusMultiplier = roundBonus.get(symbol);
    if (bonusMultiplier !== undefined && bonusMultiplier > 1) {
      return {
        source: 'roundBonus',
        multiplier: bonusMultiplier,
      };
    }
  }

  return {
    source: 'none',
    multiplier: 1,
  };
}

export function applyFinalMultiplier(
  basePayout: number,
  resolvedMultiplier: { source: 'bonusTime' | 'roundBonus' | 'none'; multiplier: number }
): number {
  if (resolvedMultiplier.source === 'none' || resolvedMultiplier.multiplier <= 0) {
    return basePayout;
  }
  return basePayout * resolvedMultiplier.multiplier;
}

export function initRoundContext(input: RoundInput): RoundContext {
  let totalBetAmount = 0;
  input.bet.forEach((amount) => {
    totalBetAmount += amount;
  });

  return {
    ...input,
    totalBetAmount,
    matches: new Map(),
    basePayouts: new Map(),
    totalBasePayout: 0,
    multipliers: new Map(),
    finalPayouts: new Map(),
    totalFinalPayout: 0,
    wasRoundWin: false,
    streakStateAfter: { ...input.streakStateBefore },
  };
}

// PURE
export function matchResultStep(ctx: RoundContext): RoundContext {
  const matches = new Map<GameSymbol, number>();
  ctx.bet.forEach((amount, symbol) => {
    if (amount > 0) {
      matches.set(symbol, getMatchCount(ctx.dice, symbol));
    }
  });

  return {
    ...ctx,
    matches,
  };
}

// PURE
export function calculateBasePayoutStep(ctx: RoundContext): RoundContext {
  const basePayouts = new Map<GameSymbol, number>();
  let totalBasePayout = 0;

  ctx.bet.forEach((amount, symbol) => {
    if (amount > 0) {
      const matchCount = ctx.matches.get(symbol) || 0;
      const payout = calculateBasePayout(amount, matchCount);
      basePayouts.set(symbol, payout);
      totalBasePayout += payout;
    }
  });

  return {
    ...ctx,
    basePayouts,
    totalBasePayout,
  };
}

// PURE
export function resolveMultiplierSourceStep(ctx: RoundContext): RoundContext {
  const multipliers = new Map<GameSymbol, { source: 'bonusTime' | 'roundBonus' | 'none'; multiplier: number }>();

  ctx.bet.forEach((amount, symbol) => {
    if (amount > 0) {
      multipliers.set(symbol, resolveMultiplierSource(ctx.streakStateBefore, ctx.roundBonus, ctx.bet, ctx.dice, symbol));
    }
  });

  return {
    ...ctx,
    multipliers,
  };
}

// PURE
export function applyFinalMultiplierStep(ctx: RoundContext): RoundContext {
  const finalPayouts = new Map<GameSymbol, number>();
  let totalFinalPayout = 0;

  ctx.bet.forEach((amount, symbol) => {
    if (amount > 0) {
      const basePayout = ctx.basePayouts.get(symbol) || 0;
      const multiplierObj = ctx.multipliers.get(symbol) || { source: 'none', multiplier: 1 };
      const finalPayout = applyFinalMultiplier(basePayout, multiplierObj);
      finalPayouts.set(symbol, finalPayout);
      totalFinalPayout += finalPayout;
    }
  });

  return {
    ...ctx,
    finalPayouts,
    totalFinalPayout,
  };
}

// PURE
export function updateStreakStateStep(ctx: RoundContext): RoundContext {
  let wasRoundWin = false;
  ctx.bet.forEach((amount, symbol) => {
    if (amount > 0) {
      const matchCount = ctx.matches.get(symbol) || 0;
      if (matchCount > 0) {
        wasRoundWin = true;
      }
    }
  });

  const streakStateAfter = updateStreakState(ctx.streakStateBefore, wasRoundWin);

  return {
    ...ctx,
    wasRoundWin,
    streakStateAfter,
  };
}

// PURE
export function updateStateStep(ctx: RoundContext): RoundResult {
  const netWinLose = ctx.totalFinalPayout - ctx.totalBetAmount;
  const balanceAfter = ctx.balanceBefore - ctx.totalBetAmount + ctx.totalFinalPayout;

  let overallSource: 'bonusTime' | 'roundBonus' | 'none' = 'none';
  let overallMultiplier = 1;

  if (ctx.streakStateBefore.bonusTimeActive) {
    overallSource = 'bonusTime';
    overallMultiplier = ctx.streakStateBefore.bonusTimeMultiplier;
  } else {
    ctx.multipliers.forEach((val, symbol) => {
      const matchCount = ctx.matches.get(symbol) || 0;
      if (matchCount > 0 && val.source === 'roundBonus' && val.multiplier > overallMultiplier) {
        overallSource = 'roundBonus';
        overallMultiplier = val.multiplier;
      }
    });
  }

  return {
    roundNumber: ctx.roundNumber,
    bet: ctx.bet,
    dice: ctx.dice,
    roundBonus: ctx.roundBonus,
    streakStateBefore: ctx.streakStateBefore,
    streakStateAfter: ctx.streakStateAfter,
    basePayout: ctx.totalBasePayout,
    bonusMultiplierSource: overallSource,
    bonusMultiplier: overallMultiplier,
    finalPayout: ctx.totalFinalPayout,
    netWinLose,
    balanceBefore: ctx.balanceBefore,
    balanceAfter,
  };
}

// PURE
export function runRoundPipeline(input: RoundInput): RoundResult {
  return pipe(
    initRoundContext(input),
    matchResultStep,
    calculateBasePayoutStep,
    resolveMultiplierSourceStep,
    applyFinalMultiplierStep,
    updateStreakStateStep,
    updateStateStep
  );
}

export function parseBetInput(input: string): { bet: Bet; error?: string } {
  const map = new Map<GameSymbol, number>();
  const tokens = input.trim().split(/\s+/);
  
  if (tokens.length === 0 || tokens[0] === "") {
    return { bet: map, error: "กรุณาระบุการเดิมพัน เช่น 'ปลา 100'" };
  }
  
  if (tokens.length % 2 !== 0) {
    return { bet: map, error: "รูปแบบไม่ถูกต้อง กรุณาพิมพ์ในรูปแบบ 'สัญลักษณ์ จำนวนเงิน' เช่น 'ปลา 100'" };
  }
  
  const VALID_SYMBOLS = new Set<GameSymbol>(['น้ำเต้า', 'ปู', 'ปลา', 'ไก่', 'เสือ', 'กุ้ง']);
  
  for (let i = 0; i < tokens.length; i += 2) {
    const symbol = tokens[i] as GameSymbol;
    const amountStr = tokens[i + 1];
    
    if (!VALID_SYMBOLS.has(symbol)) {
      return { bet: new Map(), error: `ไม่พบสัญลักษณ์ '${symbol}' (สัญลักษณ์ที่ถูกต้อง: น้ำเต้า, ปู, ปลา, ไก่, เสือ, กุ้ง)` };
    }
    
    const amount = parseInt(amountStr, 10);
    if (isNaN(amount) || amount <= 0) {
      return { bet: new Map(), error: `จำนวนเงินเดิมพัน '${amountStr}' ต้องเป็นตัวเลขมากกว่า 0` };
    }
    
    const currentBet = map.get(symbol) || 0;
    map.set(symbol, currentBet + amount);
  }
  
  return { bet: map };
}

const SYMBOLS: readonly GameSymbol[] = ['น้ำเต้า', 'ปู', 'ปลา', 'ไก่', 'เสือ', 'กุ้ง'];

// IMPURE - Calls Math.random() to simulate physical dice rolling.
// Explanation: Since random number generation produces non-deterministic output, it is impure.
// Injection Method: The generated DiceResult is passed as a pure argument into the pure pipeline (runRoundPipeline).
export function rollDice(): DiceResult {
  const roll = () => SYMBOLS[Math.floor(Math.random() * SYMBOLS.length)];
  return [roll(), roll(), roll()] as const;
}

// IMPURE - Calls Math.random() to determine random bonuses.
// Explanation: Randomness is non-deterministic, making this function impure.
// Injection Method: The generated RoundBonus map is passed as a pure argument into the pure pipeline.
export function generateRoundBonus(): RoundBonus {
  const bonusMap = new Map<GameSymbol, number>();
  const numBonusSymbols = Math.floor(Math.random() * 4); // 0, 1, 2, or 3
  const shuffledSymbols = [...SYMBOLS].sort(() => Math.random() - 0.5);
  const selectedSymbols = shuffledSymbols.slice(0, numBonusSymbols);
  const multiplierPool = [2, 4, 6, 8, 10];
  
  selectedSymbols.forEach((symbol) => {
    const randomMultiplier = multiplierPool[Math.floor(Math.random() * multiplierPool.length)];
    bonusMap.set(symbol, randomMultiplier);
  });
  
  return bonusMap;
}

// PURE
export function formatBetMap(bet: Bet): string {
  const entries: string[] = [];
  bet.forEach((amount, symbol) => {
    if (amount > 0) {
      entries.push(`${symbol}: ${amount} THB`);
    }
  });
  return entries.length > 0 ? entries.join(', ') : 'none';
}

// PURE
export function formatRoundBonus(bonus: RoundBonus): string {
  const entries: string[] = [];
  bonus.forEach((multiplier, symbol) => {
    entries.push(`${symbol} x${multiplier}`);
  });
  return entries.length > 0 ? entries.join('  |  ') : 'none';
}

  console.log(LINE_THIN);

      console.log(`  ${ICON.bye} Thanks for playing! Final balance: ${formatNumber(state.balance)} THB`);
      process.exit(0);
    }

    if (parsed.error) {
      console.log(`  ${ICON.warn} ${parsed.error}`);
      continue;

    let totalBet = 0;
    parsedBet.forEach((amount) => { totalBet += amount; });

      continue;

    if (totalBet > state.balance) {

    bet: parsedBet,

  };
  const result = runRoundPipeline(input);

  console.log(`\n${LINE_THIN}`);
  console.log(`  ${ICON.dice} Result : [ ${result.dice.join(' ]  [ ')} ]`);
  console.log(LINE_THIN);

    const basePay    = matchCount > 0 ? amount * (matchCount + 1) : 0;
    const symbolMultiplierObj = result.streakStateBefore.bonusTimeActive

      : result.roundBonus.has(symbol) && matchCount > 0
        ? { source: 'roundBonus' as const, multiplier: result.roundBonus.get(symbol)! }
        : { source: 'none'       as const, multiplier: 1 };
    const finalPay   = basePay * symbolMultiplierObj.multiplier;

    const multStr = symbolMultiplierObj.source === 'bonusTime'  ? `x${symbolMultiplierObj.multiplier} [BT]`
                  : symbolMultiplierObj.source === 'roundBonus' ? `x${symbolMultiplierObj.multiplier} [RB]`

  console.log(LINE_THIN);
  console.log(`  Summary  Round ${roundNumber}:`);
  console.log(`  ${'Total bet   :'.padEnd(18)} ${formatNumber(totalBetAmount).padStart(8)} THB`);
  console.log(`  ${'Total payout:'.padEnd(18)} ${formatNumber(result.finalPayout).padStart(8)} THB`);
  console.log(`  ${'New balance :'.padEnd(18)} ${formatNumber(result.balanceAfter).padStart(8)} THB`);
  
  const faces = result.netWinLose >= 0 ? WIN_FACES : LOSE_FACES;
  const selectedFace = getRandomFace(faces);
  const outcomeMsg = formatOutcomeMessage(result.netWinLose, selectedFace);
  console.log(`  Outcome     : ${outcomeMsg}`);

  // --- Streak / Bonus Time Notification ---
  console.log(LINE_THIN);
  if (!result.streakStateBefore.bonusTimeActive && result.streakStateAfter.bonusTimeActive) {
    console.log(`  ${ICON.ok} 3-win streak! >> BONUS TIME activates next round (x2 all bets)`);
  } else if (result.streakStateBefore.bonusTimeActive) {
    if (result.streakStateAfter.bonusTimeActive) {
      console.log(`  ${ICON.bonusTime} Streak continues! Multiplier -> x${result.streakStateAfter.bonusTimeMultiplier} next round`);
    } else {
      console.log(`  ${ICON.warn} Lost during Bonus Time — streak reset, back to normal mode`);
    }
  }

  // --- Log to file ---
  const logEntry = `[Round ${result.roundNumber}] Date: ${new Date().toISOString()} | BalanceBefore: ${result.balanceBefore} | TotalBet: ${totalBetAmount} | Bets: ${formatBetMap(result.bet)} | Dice: ${result.dice.join(',')} | BonusSource: ${result.bonusMultiplierSource} x${result.bonusMultiplier} | Payout: ${result.finalPayout} | Net: ${result.netWinLose} | BalanceAfter: ${result.balanceAfter} | StreakBefore: (wins:${result.streakStateBefore.consecutiveWins}, active:${result.streakStateBefore.bonusTimeActive}, mult:${result.streakStateBefore.bonusTimeMultiplier}) | StreakAfter: (wins:${result.streakStateAfter.consecutiveWins}, active:${result.streakStateAfter.bonusTimeActive}, mult:${result.streakStateAfter.bonusTimeMultiplier})`;
  writeLogToFile(logEntry);

  const nextState: GameState = {
    balance: result.balanceAfter,
    streakState: result.streakStateAfter,
    history: [...state.history, result],
  };

  if (nextState.balance <= 0) {
    console.log(`\n  ${ICON.warn} Balance is 0. Game over!`);
    const action = await promptUser(`  Reload 1,000 THB and continue? (y/n): `);
    if (action.toLowerCase() === 'y') {
      const resetState: GameState = {
        balance: 1000,
        streakState: { consecutiveWins: 0, bonusTimeActive: false, bonusTimeMultiplier: 0 },
        history: nextState.history,
      };
      await gameLoop(resetState, roundNumber + 1);
    } else {
      console.log(`  ${ICON.bye} Thanks for playing!`);
      process.exit(0);
    }
  } else {
    const action = await promptUser(`\n  Press Enter for next round, or type "exit" to quit: `);
    if (action.toLowerCase() === 'exit') {
      console.log(`  ${ICON.bye} Final balance: ${formatNumber(nextState.balance)} THB. Goodbye!`);
      process.exit(0);
    }
    await gameLoop(nextState, roundNumber + 1);
  }
}
